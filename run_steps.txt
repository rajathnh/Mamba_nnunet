Create docker image:

docker run --name mamba3-sandbox --gpus all -it --ipc=host --shm-size=8g \
    -v ~/mamba_project/sandbox:/workspace \
    pytorch/pytorch:2.11.0-cuda12.6-cudnn9-devel /bin/bash
